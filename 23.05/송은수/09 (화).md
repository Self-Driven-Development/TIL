void writeRankFile() {

	if (sn == score_number) return;
	//2. 랭킹 정보들의 수를 "rank.txt"에 기록
	fp = fopen("rank.txt", "w");
	fprintf(fp, "%d\n", score_number);
	//3. 탐색할 노드가 더 있는지 체크하고 있으면 다음 노드로 이동, 없으면 종료

	fclose(fp);
}
